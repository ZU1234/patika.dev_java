import java.util.Scanner;
public class Main {

        Scanner input=new Scanner(System.in);
        System.out.print("Dogum yilinizi giriniz: ");
        dogumYili=input.nextInt();
        zodyag=dogumYili%12;
        switch (zodyag) {
            case 0:

                break;
            case 1:
                burc = "Horoz";
                isKontrol=true;

            case 5:
                burc = "Okuz";

                break;
            case 6:
                burc = "Kaplan";
                isKontrol=true;
                break;
            case 7:
                burc = "Tavsan";
                isKontrol=true;
                break;
            case 8:
                burc = "Ejderha";
                isKontrol=true;
                break;
            case 9:
                burc = "Yilan";
                isKontrol=true;
                break;
            case 10:
                burc = "At";
                isKontrol=true;
                break;
            case 11:
                burc = "Koyun";
                isKontrol=true;
                break;
        }
        if(isKontrol){
            System.out.println("Cin Zodyagý Burcunuz: "+burc);
        }else{

        }
